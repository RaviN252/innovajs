Comparision operator

        1. == (equals to value)
        let num14 = 5;
        console.log(num14 == num15)   true 

        let num15 = "5";
        console.log(num14 == num15)   true  

        2. === ( equals to value and type)

        let num14 = 5;
        console.log(num14 === num15)   true 
        
        let num15 = "5";
        console.log(num14 === num15)  false
        because it checks both the value and datatypes


        3. != (not equals)
        let num14 = 5;
        let num15 = "5";
        console.log(num14 != num15);   false
        
        4. !== (not equals to value and type)
        let num14 = 5;
        let num15 = "5";
        console.log(num14 !== num15);   true

        5. > (greater than)

        let num14 = 5;
        let num15 = 6;
        console.log(num14 > num15);  false
        6. < (less than )

        let num14 = 5;
        let num15 = 6;
        console.log(num14 < num15);  true

        7. >= ( grater than or equals to)
        8. <= ( lesser than or equals to)

        checks the both condition
